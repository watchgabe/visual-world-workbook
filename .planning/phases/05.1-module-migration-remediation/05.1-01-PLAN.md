const [isGenerating, setIsGenerating] = useState<string | null>(null)

// Generation handler
async function handleGenerate(fieldKey: string, prompt: string) {
  setIsGenerating(fieldKey)
  try {
    const res = await fetch('/api/claude', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt, maxTokens: 600 }),
    })
    const data = await res.json()
    if (data.error) throw new Error(data.error)
    ;(setValue as (k: string, v: string) => void)(fieldKey, data.text || '')
  } catch {
    // error handling
  } finally {
    setIsGenerating(null)
  }
}

// Button rendering
<button
  onClick={() => handleGenerate('field_key')}
  disabled={isGenerating === 'field_key'}
  style={{
    opacity: isGenerating === 'field_key' ? 0.6 : 1,
    cursor: isGenerating === 'field_key' ? 'not-allowed' : 'pointer',
  }}
>
  {isGenerating === 'field_key' ? 'Generating...' : '✦ Generate'}
</button>
