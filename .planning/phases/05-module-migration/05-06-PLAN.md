**Data fetching pattern (per D-07 — real-time query on page load):**

```typescript
'use client'

import { useEffect, useState } from 'react'
import { createClient } from '@/lib/supabase/client'
import { useAuth } from '@/context/AuthContext'
import { MODULES, MODULE_SECTIONS } from '@/lib/modules'
import type { ModuleSlug } from '@/types/database'

type ResponsesByModule = Record<string, Record<string, unknown>>

export default function PlaybookPage() {
  const { user } = useAuth()
  const [responses, setResponses] = useState<ResponsesByModule>({})
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    if (!user) return
    const supabase = createClient()
    supabase
      .from('blp_responses')
      .select('module_slug, responses')
      .eq('user_id', user.id)
      .then(({ data }) => {
        const byModule: ResponsesByModule = {}
        for (const row of data ?? []) {
          byModule[row.module_slug] = row.responses as Record<string, unknown>
        }
        setResponses(byModule)
        setLoading(false)
      })
  }, [user])

  if (loading) return <PlaybookSkeleton />

  return (
    <div style={{ maxWidth: '700px', margin: '0 auto', padding: '2rem 2rem 5rem' }}>
      {/* Cover section — matches old playbook.html pb-cover */}
      {/* Metadata row — user info */}
      {/* Chapter sections — one per module */}
      {/* Print button */}
    </div>
  )
}
```

**Rendering structure (from old HTML):**

The old playbook.html renders chapters for each module:
1. **Cover section** — large title "THE PLAYBOOK", eyebrow text, brand name from responses
2. **Metadata row** — 2x2 or 4-column grid with brand info (name, date, tagline, etc.)
3. **Section nav row** — links to chapters (anchor-based in old HTML; can keep as smooth-scroll anchors)
4. **Chapters** — one per module (Brand Foundation, Visual World, Content, Launch):
   - Chapter header with number and title
   - Grid of field cards: label + value
   - Each card: `background: 'var(--surface)'; border: '1px solid var(--border)'; borderRadius: 'var(--radius-md)'; padding: '.9rem 1rem'`
   - Label: `fontSize: '9px'; fontWeight: 700; textTransform: 'uppercase'; letterSpacing: '.1em'; color: 'var(--orange)'`
   - Value: `fontSize: '13px'; color: 'var(--text)'; lineHeight: 1.6`
   - Empty values: `color: 'var(--dimmer)'; fontStyle: 'italic'` with text "Not yet completed"

/* ── Print styles for Playbook ── */
@media print {
  /* Hide navigation and UI chrome */
  aside,
  .mobile-topbar,
  .print-btn {
    display: none !important;
  }

  /* Reset main content to full width */
  main {
    margin-left: 0 !important;
    padding: 0 !important;
  }

  /* Clean background for print */
  body {
    background: #fff !important;
    color: #000 !important;
  }

  /* Ensure cards print with borders visible */
  * {
    -webkit-print-color-adjust: exact !important;
    print-color-adjust: exact !important;
  }

  /* Page breaks — keep chapters together */
  .pb-chapter {
    break-inside: avoid;
    page-break-inside: avoid;
  }
}
