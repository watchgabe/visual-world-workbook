**1. Create `src/app/(app)/modules/[slug]/layout.tsx` — module-level layout with SectionNav:**

This is a client component (`'use client'`) that:
- Reads `slug` from `useParams()` (Next.js 15 App Router)
- Looks up `MODULE_SECTIONS[slug]` from `src/lib/modules.ts`
- If the module has no sections (welcome, playbook), renders `{children}` only
- If the module has sections, renders `SectionNav` above `{children}`
- Uses `usePathname()` to determine `activeIndex` by finding which section slug is in the current URL path
- `onSectionChange` calls `router.push(\`/modules/${slug}/${sections[index].slug}\`)`
- SectionNav sections array maps SectionDef[] to `{ name, complete: false }` (completion wiring deferred — each section page manages its own state; SectionNav shows navigation only for now)

```typescript
'use client'

import { useParams, usePathname, useRouter } from 'next/navigation'
import { MODULE_SECTIONS } from '@/lib/modules'
import { SectionNav } from '@/components/workshop/SectionNav'
import type { ModuleSlug } from '@/types/database'

export default function ModuleLayout({ children }: { children: React.ReactNode }) {
  const params = useParams()
  const pathname = usePathname()
  const router = useRouter()

  const slug = params.slug as ModuleSlug
  const sections = MODULE_SECTIONS[slug]

  if (!sections || sections.length === 0) {
    return <>{children}</>
  }

  const activeIndex = sections.findIndex(s => pathname.endsWith(`/${s.slug}`))

  const handleSectionChange = (index: number) => {
    router.push(`/modules/${slug}/${sections[index].slug}`)
  }

  return (
    <div>
      <SectionNav
        sections={sections.map(s => ({ name: s.name, complete: false }))}
        activeIndex={activeIndex >= 0 ? activeIndex : 0}
        onSectionChange={handleSectionChange}
      />
      {children}
    </div>
  )
}
```

interface ModulePageProps {

export default async function ModulePage({ params }: ModulePageProps) {

}

**3. Create `src/app/(app)/modules/[slug]/[section]/page.tsx` — dynamic section router:**

CRITICAL: This file is the single routing entry point for ALL module sections across ALL modules. It uses `next/dynamic` to lazily import section components based on slug and section params. This means Plans 02-05 only need to create section component files in `src/components/sections/{module}/` — they do NOT need to modify this routing file.

```typescript
import dynamic from 'next/dynamic'
import { notFound } from 'next/navigation'
import { MODULES, MODULE_SECTIONS } from '@/lib/modules'
import type { ModuleSlug } from '@/types/database'

interface SectionPageProps {
  params: Promise<{ slug: string; section: string }>
}

// Registry mapping slug/section to lazy-loaded component paths
// Uses next/dynamic for code splitting — each section component is loaded on demand
function getSectionComponent(slug: string, section: string) {
  try {
    return dynamic(
      () => import(`@/components/sections/${slug}/${section}`),
      { loading: () => <div style={{ padding: '2rem', color: 'var(--dim)' }}>Loading section...</div> }
    )
  } catch {
    return null
  }
}

export default async function SectionPage({ params }: SectionPageProps) {
  const { slug, section } = await params

  // Validate slug exists
  const mod = MODULES.find(m => m.slug === slug)
  if (!mod) notFound()

  // Validate section exists for this module
  const sections = MODULE_SECTIONS[slug as ModuleSlug]
  if (!sections?.find(s => s.slug === section)) notFound()

  // Load section component dynamically
  const Component = getSectionComponent(slug, section)
  if (!Component) notFound()

  return <Component />
}
```

This pattern uses file-convention-based dynamic imports: each section component must be the default export from `src/components/sections/{slug}/{section-slug}.tsx` (or `/index.tsx`). Plans 02-05 create these files without touching the router.

**IMPORTANT:** The file naming convention for section components must match the section slug exactly. E.g., `src/components/sections/brand-foundation/brand-journey.tsx` for the `brand-journey` section. Use kebab-case file names matching section slugs.
