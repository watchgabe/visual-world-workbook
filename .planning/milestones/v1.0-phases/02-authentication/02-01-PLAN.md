```typescript
import { createServerClient } from '@supabase/ssr'
import { cookies } from 'next/headers'
import type { Database } from '@/types/database'

export async function createClient() {
  const cookieStore = await cookies()
  return createServerClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return cookieStore.getAll() },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            )
          } catch {
            // Server components cannot set cookies — middleware handles refresh
          }
        },
      },
    }
  )
}
```

**1. Create `src/app/auth/callback/route.ts`:**

Export a `GET` route handler:
- Extract `searchParams` and `origin` from `new URL(request.url)`.
- Read `code`, `next` (default to `'/'`), `error`, and `error_description` from searchParams.
- If `error` param exists: redirect to `/login?error=link_expired` (per D-13 — friendly error handling).
- If `code` exists: call `await createClient()` from `@/lib/supabase/server`, then `await supabase.auth.exchangeCodeForSession(code)`.
  - On success (no error): redirect to `new URL(next, origin)` — this is the return URL from D-06.
  - On error: redirect to `/login?error=link_expired`.
- If no code and no error: redirect to `/login?error=unknown`.

Use `NextResponse.redirect()` for all redirects. Import `createClient` from `@/lib/supabase/server`.

**2. Create `src/context/AuthContext.tsx`:**

Mark as `'use client'`.

Define interface:
```typescript
interface AuthContextValue {
  user: User | null
  loading: boolean
  signOut: () => Promise<void>
}
```

Import `User` type from `@supabase/supabase-js`. Import `createClient` from `@/lib/supabase/client`.

Create `AuthContext` with `createContext<AuthContextValue>({ user: null, loading: true, signOut: async () => {} })`.

Create `AuthProvider({ children }: { children: React.ReactNode })`:
- State: `user` (User | null, initial null), `loading` (boolean, initial true).
- In `useEffect`:
  - Call `supabase.auth.getUser()` to set initial user and loading=false.
  - Call `supabase.auth.onAuthStateChange((_event, session) => setUser(session?.user ?? null))`.
  - Return cleanup: `subscription.unsubscribe()`.
- `signOut` function: `await supabase.auth.signOut()` then `window.location.href = '/login'` (hard redirect per Pitfall 4 — avoid stale router cache).

Export `AuthProvider` and `useAuth = () => useContext(AuthContext)`.
